#include <stdio.h>

int main(){
	int t;
    scanf("%d", &t);

    while (t--) {
        int G, P, N;
        scanf("%d %d", &G, &P);
        scanf("%d", &N);

        int solve1 = 0, solve2 = 0;

        for (int i = 0; i < N; i++) {
            int x, y;
            scanf("%d %d", &x, &y);
            if (x == 1) solve1++;
            if (y == 1) solve2++;
        }

        long long cost1 = (long long)solve1 * G + (long long)solve2 * P;
        long long cost2 = (long long)solve1 * P + (long long)solve2 * G;

        printf("%lld\n", cost1 < cost2 ? cost1 : cost2);
    }

    return 0;
}
