#include <stdio.h>
#include <ctype.h>

int main() {
    char s[100];
    scanf("%s", s);

    for (int i = 0; s[i] != '\0'; i++) {
        if (islower(s[i]))
            s[i] = toupper(s[i]);
        else
            s[i] = tolower(s[i]);
    }

    printf("%s\n", s);
    return 0;
}
